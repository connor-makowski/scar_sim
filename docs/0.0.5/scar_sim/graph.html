 18    def update_graphs(self, obj: Arc | Node):
 19        """
 20        Updates the graph representations with the provided Arc or Node object.
 21
 22        This should be called when an Arc or Node's processing parameters are changed to ensure the graph representations remain accurate.
 23
 24        Any cashflow that is positive will be treated as zero since cashflows in the graph are only used to represent costs (negative cashflows).
 25
 26        If an arc is symmetric, the graph will be updated in both directions.
 27
 28        Required Arguments:
 29
 30        - obj (Arc | Node): The Arc or Node object to update the graphs with.
 31
 32        Returns:
 33
 34        - None
 35        """
 36        if isinstance(obj, Arc):
 37            self.time_graph[obj.origin_node.outbound_graph_id][
 38                obj.destination_node.inbound_graph_id
 39            ] = max(float(obj.processing_time_avg), 0)
 40            self.cashflow_graph[obj.origin_node.outbound_graph_id][
 41                obj.destination_node.inbound_graph_id
 42            ] = max(-float(obj.processing_cashflow_per_unit), 0)
 43            self.arc_obj_graph[obj.origin_node.outbound_graph_id][
 44                obj.destination_node.inbound_graph_id
 45            ] = obj
 46            if obj.is_symmetric:
 47                self.time_graph[obj.destination_node.outbound_graph_id][
 48                    obj.origin_node.inbound_graph_id
 49                ] = max(float(obj.processing_time_avg), 0)
 50                self.cashflow_graph[obj.destination_node.outbound_graph_id][
 51                    obj.origin_node.inbound_graph_id
 52                ] = max(-float(obj.processing_cashflow_per_unit), 0)
 53                self.arc_obj_graph[obj.destination_node.outbound_graph_id][
 54                    obj.origin_node.inbound_graph_id
 55                ] = obj
 56        elif isinstance(obj, Node):
 57            self.time_graph[obj.inbound_graph_id] = {
 58                obj.outbound_graph_id: max(float(obj.processing_time_avg), 0)
 59            }
 60            self.cashflow_graph[obj.inbound_graph_id] = {
 61                obj.outbound_graph_id: max(
 62                    -float(obj.processing_cashflow_per_unit), 0
 63                )
 64            }

 66    def add_object(self, obj: Node | Arc) -> Node | Arc:
 67        """
 68        Adds a Node or Arc object to the graph representations.
 69
 70        Nodes are actually represented as two nodes in the graph (inbound and outbound) connected by an arc that models processing time and cashflow.
 71
 72        Graphs are updated accordingly.
 73
 74        Required Arguments:
 75
 76        - obj (Node | Arc): The Node or Arc object to add to the graphs.
 77        """
 78        if isinstance(obj, Node):
 79            obj.inbound_graph_id = len(self.time_graph)
 80            obj.outbound_graph_id = obj.inbound_graph_id + 1
 81            self.time_graph += [dict(), dict()]
 82            self.cashflow_graph += [dict(), dict()]
 83            self.arc_obj_graph += [dict(), dict()]
 84        if isinstance(obj, Node | Arc):
 85            self.update_graphs(obj)
 86        return obj

 88    def get_path_weight(
 89        self, path: list[int], graph: Literal["cashflow", "time"]
 90    ) -> float:
 91        """
 92        Given a path (list of graph IDs), calculates the total weight of the path based on the specified graph type.
 93
 94        Required Arguments:
 95
 96        - path (list[int]): A list of graph IDs representing the path.
 97        - graph (Literal['cashflow', 'time']): The type of graph to use for weight calculation.
 98
 99        Returns:
100
101        - float: The total weight of the path.
102            - Note: Since cashflows are stored as negative values, the returned cashflow weight will be adjusted back to its original sign for consistency.
103        """
104        graph_obj = (
105            self.cashflow_graph if graph == "cashflow" else self.time_graph
106        )
107        weight_sum = 0.0
108        for i in range(len(path) - 1):
109            origin = path[i]
110            destination = path[i + 1]
111            weight_sum += graph_obj[origin][destination]
112        return weight_sum if graph == "time" else -weight_sum

114    def get_optimal_path(
115        self,
116        origin_node: Node,
117        destination_node: Node,
118        graph: Literal["cashflow", "time"],
119    ) -> list[int]:
120        """
121        Calculates the optimal path between the specified origin and destination nodes based on the specified graph type.
122
123        Required Arguments:
124
125        - origin_node (Node): The starting node for the path.
126        - destination_node (Node): The ending node for the path.
127        - graph (Literal['cashflow', 'time']): The type of graph to use for path calculation.
128
129        Returns:
130
131        - list[int]: A list of graph IDs representing the optimal path.
132        """
133        graph_obj = (
134            self.cashflow_graph if graph == "cashflow" else self.time_graph
135        )
136        return SCGraph.dijkstra_makowski(
137            graph_obj,
138            origin_node.inbound_graph_id,
139            destination_node.inbound_graph_id,
140        )["path"]
