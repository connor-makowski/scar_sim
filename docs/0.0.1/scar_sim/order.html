 41        """
 42        Starts the Order's processing within the simulation.

 94    def __set_planned_path__(
 95        self, planned_path: list[int], initial=False
 96    ) -> None:
 97        """
 98        An internal method to set the planned path for the Order.
 99
100        This is expected to be called internally when the Order is created and when rerouting is performed.
101
102        Required Arguments:
103
104        - planned_path (list[int]): The new planned path as a list of graph IDs.
105        - initial (bool): Whether this is the initial setting of the planned path. Defaults to False.
106
107        Raises:
108
109        - ValueError: If the new path does not match the current path up to the current location (when not initial).
110        - AssertionError: If the planned path does not start at the origin node or end at the destination node.
111            - Note: More specifically, the planned path must start with the origin node's inbound graph ID and end with the destination node's inbound graph ID.
112
113        Returns:
114        - None
115        """
116        if not initial:
117            if (
118                planned_path[: self.__current_path_idx__ + 1]
119                != self.__planned_path__[: self.__current_path_idx__ + 1]
120            ):
121                raise ValueError(
122                    "New path must match current path up to current location"
123                )
124        assert (
125            planned_path[0] == self.origin_node.inbound_graph_id
126        ), "Planned path must start at origin node"
127        assert (
128            planned_path[-1] == self.destination_node.inbound_graph_id
129        ), "Planned path must end at destination node"
130        self.__planned_path__ = planned_path

132    def __next__(self, status: str) -> None:
133        """
134        An internal method to progress the Order to the next status in its lifecycle.
135
136        This method is expected to be stored as an event in the simulation's event queue and fired off at the appropriate times.
137
138        Required Arguments:
139
140        - status (str): The current status of the Order. Must be one of "started", "shipped", "arrived", or "completed".
141
142        Raises:
143
144        - ValueError: If the status is unknown or if the current object does not match the expected type for the status transition.
145
146        Returns:
147
148        - None
149        """
150        # Log this item into the Order history
151        self.history.append(
152            {
153                "time": self.__simulation__.current_time(),
154                "time_delta": round(
155                    self.__simulation__.current_time() - self.__prev_time__, 3
156                ),
157                "order_id": self.id,
158                "current_obj_id": self.__current_object__.id,
159                "meta": self.__current_object__.get_metadata(
160                    **self.inject_metadata()
161                ),
162                "status": status,
163                "cashflow": 0.0,
164            }
165        )
166        self.__current_path_idx__ += 1
167        self.__prev_time__ = self.__simulation__.current_time()
168
169        if status == "started":
170            # Validate that we are at a Facility that can process orders
171            if not isinstance(self.__current_object__, Facility):
172                raise ValueError(
173                    "Current object must be a Facility when started"
174                )
175            # Perform any logic at the origin facility to process the order (i.e., remove from capacity)
176            self.__current_object__.order_placed(self)
177            # Set up for shipping
178            next_status = "shipped"
179        elif status == "shipped":
180            # If the current object is a Facility, we need to ship the order (i.e., remove from inventory)
181            if isinstance(self.__current_object__, Facility):
182                self.__current_object__.order_shipped(self)
183
184            # Pay for processing an order when it is shipped from a node
185            self.set_current_cashflow(
186                self.__current_object__.get_cashflow(units=self.units)
187            )
188            # Given the planned path, set the current object to the next planned Arc
189            next_graph_id = self.__planned_path__[self.__current_path_idx__]
190            self.__current_object__ = self.__simulation__.graph.arc_obj_graph[
191                self.__current_object__.outbound_graph_id
192            ][next_graph_id]
193            next_status = "arrived"
194        elif status == "arrived":
195            if not isinstance(self.__current_object__, Arc):
196                raise ValueError("Current object must be an Arc when arriving")
197            # Pay for the transportation when a unit arrives at the destination node
198            self.set_current_cashflow(
199                self.__current_object__.get_cashflow(units=self.units)
200            )
201            # Set the current object to the destination Node of the Arc
202            self.__current_object__ = self.__current_object__.destination_node
203            if not isinstance(self.__current_object__, Node):
204                raise ValueError("Next object must be a Node when arriving")
205            # Consider rerouting the order
206            should_reroute, new_path = self.consider_reroute()
207            if should_reroute:
208                self.__set_planned_path__(new_path)
209            if isinstance(self.__current_object__, Facility):
210                # Only fire off the order arrived event if we are at a Facility and a reroute has not changed the path away from this Facility
211                # Fire off the order arrived event at the Facility for processing (i.e., add to inventory)
212                if len(self.__planned_path__) == self.__current_path_idx__:
213                    # We are at the end of the path
214                    self.__current_object__.order_arrived(self)
215                elif (
216                    self.__planned_path__[self.__current_path_idx__ + 1]
217                    == self.__current_object__.outbound_graph_id
218                ):
219                    self.__current_object__.order_arrived(self)
220            next_status = (
221                "completed"
222                if self.__current_object__.id == self.destination_node.id
223                else "shipped"
224            )
225        elif status == "completed":
226            # Validate that we are at a Facility that can receive Orders
227            if not isinstance(self.__current_object__, Facility):
228                raise ValueError(
229                    "Current object must be a Facility when completed"
230                )
231            # Fire off the order completed event at the Facility for processing (i.e., add to capacity)
232            self.__current_object__.order_completed(self)
233            # When called with "completed", we do not schedule any further events
234            return
235        else:
236            raise ValueError(f"Unknown status: {status}")
237
238        self.__simulation__.add_event(
239            time_delta=(
240                self.__current_object__.get_processing_time()
241                if next_status != "completed"
242                else 0.0
243            ),
244            func=self.__next__,
245            kwargs={"status": next_status},
246        )
